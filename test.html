const Text = require('proglib').Text;
const StringBuilder = Text.StringBuilder;

// Инициализация StringBuilder со строкой "ABC".
let sb = new StringBuilder("ABC");

// Добавление трёх символов (D, E и F) в конец StringBuilder.
sb.Append([ 'D', 'E', 'F' ]);

// Добавление строки формата в конец StringBuilder.
sb.AppendFormat("GHI{0}{1}", 'J', 'k');

// Отображение количества символов в StringBuilder и его строки.
console.log(`${sb.length}, chars: ${sb.toString()}`);

// Вставка строки в начало StringBuilder.
sb.Insert(0, "Alphabet: ");

// Замена всех строчных букв k на прописные.
sb.Replace('k', 'K');

// Отображение количества символов в StringBuilder и его строки.
console.log(`${sb.length}, chars: ${sb.toString()}`);

// Этот код выдаёт следующие выходные данные.
//
// 11, chars: ABCDEFGHIJk
// 21, chars: Alphabet: ABCDEFGHIJK
